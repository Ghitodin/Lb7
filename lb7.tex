\begin{tcolorbox}[rounded corners, top=1mm, bottom=1mm]
\textbf{Завдання 3.} Що виведе наступний код:
\begin{minted}[bgcolor=gray!20, fontsize=\scriptsize, linenos, numbersep=3pt, frame=lines, framesep=1mm]{python}

        b = b + a
    else:

print(b)
\end{minted}

\begin{tcolorbox}[rounded corners, top=1mm, bottom=1mm]
